3. 守卫使用方式

装饰器控制鉴权行为
// 公开接口 — 跳过所有鉴权
@Public()
@Get('getCaptcha')
getCaptcha() {}
// 仅需登录 — 无额外装饰器
@Get('getUserInfo')
getUserInfo() {}
// 登录 + 指定权限
@Permissions('user:list')
@Get('getUserList')
getUserList() {}
